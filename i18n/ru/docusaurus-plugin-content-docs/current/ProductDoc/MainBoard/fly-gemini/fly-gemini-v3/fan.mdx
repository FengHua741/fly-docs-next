# Разъем для встроенного вентилятора 5В

На Gemini предусмотрен разъем для встроенного вентилятора 5В, предназначенный для охлаждения основного контроллера. Схема подключения представлена на следующем изображении:

    <ImageView image={require('@site/docs/ProductDoc/MainBoard/fly-gemini/fly-gemini-v3/img/corefan.webp').default} size="80%" align="center" />

## Пример конфигурации

    ```cfg
    [mcu host]                      # FlyOS уже автоматически сгенерировал linux MCU, определение которого достаточно для использования            
    serial: /tmp/klipper_host_mcu 

    [temperature_sensor FLY-π]      # Температура верхнего контроллера
    sensor_type: temperature_host


    [temperature_fan core_fan]      # Вентилятор охлаждения верхнего контроллера
    pin: host:gpiochip1/gpio103     # Ножка для подключения вентилятора верхнего контроллера
    max_power: 1.0
    sensor_type: temperature_host   # Указывается, что источник температуры - верхний контроллер
    control:watermark               # Метод управления
    target_temp: 48                 # Температура запуска вентилятора
    min_temp: 0                     # Минимальная температура, при которой будет выдано сообщение об ошибке
    max_temp: 90                    # Максимальная температура, при которой будет выдано сообщение об ошибке
    off_below: 0.10
    kick_start_time: 0.50
    max_speed: 0.8                  # Максимальная скорость вентилятора (80% от полной мощности)
    min_speed: 0.3                  # Минимальная скорость вентилятора (30% от полной мощности)
    ```
