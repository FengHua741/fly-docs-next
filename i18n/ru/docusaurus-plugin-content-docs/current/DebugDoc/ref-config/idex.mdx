Копирование или зеркальное отображение макросов требует ссылки на соответствующий код в срезке

* ACTIVATE_COPY_MODE Копировать режим
* ACTIVATE_MIRROR_MODE Режим отражения

```cfg
[gcode_macro PARK_extruder1]
gcode:
    SAVE_GCODE_STATE NAME=park1
    G90
    G1 X200
    RESTORE_GCODE_STATE NAME=park1

[gcode_macro T1]
gcode:
    PARK_{printer.toolhead.extruder}
    ACTIVATE_EXTRUDER EXTRUDER=extruder1
    SET_DUAL_CARRIAGE CARRIAGE=1
    SET_GCODE_OFFSET Y=15

# Вспомогательный скрипт для активации режима копирования
[gcode_macro ACTIVATE_COPY_MODE]
gcode:
    SET_DUAL_CARRIAGE CARRIAGE=0 MODE=PRIMARY
    G1 X0
    ACTIVATE_EXTRUDER EXTRUDER=extruder
    SET_DUAL_CARRIAGE CARRIAGE=1 MODE=PRIMARY
    G1 X100
    SET_DUAL_CARRIAGE CARRIAGE=1 MODE=COPY
    SYNC_EXTRUDER_MOTION EXTRUDER=extruder1 MOTION_QUEUE=extruder

# Вспомогательный скрипт для активации режима отражения
[gcode_macro ACTIVATE_MIRROR_MODE]
gcode:
    SET_DUAL_CARRIAGE CARRIAGE=0 MODE=PRIMARY
    G1 X0
    ACTIVATE_EXTRUDER EXTRUDER=extruder
    SET_DUAL_CARRIAGE CARRIAGE=1 MODE=PRIMARY
    G1 X200
    SET_DUAL_CARRIAGE CARRIAGE=1 MODE=MIRROR
    SYNC_EXTRUDER_MOTION EXTRUDER=extruder1 MOTION_QUEUE=extruder
```

[gcode_macro T0]
gcode:
	RESPOND MSG="Выбор инструмента T0"
	home_if_not
	{% if printer.toolhead.extruder != "extruder" %}
	    PARK_{printer.toolhead.extruder}
	{% endif %}
    ACTIVATE_EXTRUDER EXTRUDER=extruder
    SET_DUAL_CARRIAGE CARRIAGE=0
	# запуск настройки диапазона только один раз, чтобы не неправильно установить его после того, как целевая температура T1 установлена в ноль
	{% if  printer["gcode_macro set_stepper_x_range"].range_set  == 0  %}
		RESPOND MSG="Диапазон движения левого экструдера по оси X еще не установлен, устанавливаем его сейчас"
		set_stepper_x_range
	{% endif %}
	x_xoffset

	home_if_not

	    PARK_{printer.toolhead.extruder}
	{% endif %}

[gcode_macro PRIME_LINE]
gcode: 
    RESPOND MSG="Печать линии для первичной подготовки"

    {% set x = params.X|default(2)|float %}
    {% set y = params.Y|default(70)|float %}
	{% set z = params.Z|default(0.2)|float %}
    {% set xline = params.XLINE|default(0.8)|float %}
    {% set yline = params.YLINE|default(120)|float %}
	{% set eMultiplier = 0.15 * z  %}
    SAVE_GCODE_STATE NAME=prime_line
 
	M117 Priming Line

    G90 ; XYZ в абсолютный режим
	M83 ; установить E в относительный режим
    # перемещение оси Z 
    G1 Z0.3 F3000 ;Поднять ось Z
    # переместиться в начальную позицию для первичной подготовки 
	G0 Y{y} Z{z}
    G0 X{x} ;Переместиться в начальную позицию. Сделать квадратное движение, чтобы избежать столкновения с углами основания
	G1 Y{y + yline} E{eMultiplier * yline} ;Нарисовать первую линию
    G1 X{x + xline} E{eMultiplier * xline} ;Переместиться немного в сторону
    G1 Y{y + 20} E{eMultiplier * (yline - 20)} ;Нарисовать вторую линию, но не вернуться к исходной позиции, чтобы избежать снятия излишков
#	G1 E-0.5 F300
    G1 Z0.5 F3000 ;Поднять ось Z
	RESTORE_GCODE_STATE NAME=prime_line

[gcode_macro SEARCH_VARS]
gcode:
    {% set search = params.S|lower %}
    {% set ns = namespace() %}
    {% for item in printer  %}
        {% if ' ' in item %}
            {% set ns.path = ['printer', "['%s']" % (item), ''] %}
        {% else %}
            {% set ns.path = ['printer.', item, ''] %}   
        {% endif %} 

        {% if search in ns.path|lower %}
            { action_respond_info(ns.path|join) }
        {% endif %} 

        {% if printer[item].items() %}
            {% for childkey, child in printer[item].items() recursive %}
                {% set ns.path = ns.path[:loop.depth|int + 1] %}

                {% if ' ' in childkey %}
                    {% set null = ns.path.append("['%s']" % (childkey)) %}
                {% else %}
                    {% set null = ns.path.append(".%s" % (childkey)) %}
                {% endif %} 

                {% if child is mapping  %}
                    { loop(child.items()) }
                {% else %}
                    {% if search in ns.path|lower %}
                        { action_respond_info("%s : %s" % (ns.path|join, child)) }   
                    {% endif %} 
                {% endif %} 
                
            {% endfor %}
        {% endif %} 
    {% endfor %}
